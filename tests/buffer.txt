/* Buffer Example */

automata {
    Buff = (in -> out -> Buff).
    B2 = (a:Buff/{x/a.out} || b:Buff/{x/b.in})\{x}.
    B3 = abs(B2).
    B4 = simp(B3).
}

petrinet {
    PNBuff = (in -> out -> PNBuff).
    PNB2 = (a:PNBuff/{x/a.out} || b:PNBuff/{x/b.in})\{x}.
    PNB3 = abs(PNB2).
    PNB4 = simp(PNB3).
}

operation {
    Buff ~ automata(PNBuff).
    B2 ~ automata(PNB2).
    B3 ~ automata(PNB3).
    B4 ~ automata(PNB4).
}
